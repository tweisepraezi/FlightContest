OSM data (as gis)
-----------------

    # Download osm data (Saarland)
    cd ~/data
    mkdir osm && cd osm
    wget https://download.geofabrik.de/europe/germany/saarland-latest.osm.pbf

    # Create database *gis_new*
    dropdb gis_new
    createdb gis_new
    psql -d gis_new -c 'CREATE EXTENSION postgis;'

    # Load OSM data into OSM database (~/data/osm/*.pbf -> gis_new, 24h)
    osm2pgsql -d gis_new --slim --drop --bbox 6.24522242525,48.9348891599,7.83147757475,49.9660696535 --flat-nodes=osmnodes.dat --number-processes 10 --output pgsql --style ~/fcmaps/osmtools/fcmaps.style ~/data/osm/*.osm.pbf
        # https://osm2pgsql.org/doc/man/osm2pgsql-2-2-0.html

    # Preprocessing
    cd ~/fcmaps/osmtools/
    bash run_presteps.sh
    
    # Activate new databases *gis_eu* and *lowzoom_eu*
    dropdb gisold_eu
    dropdb lowzoomold_eu
    psql -d postgres -c 'ALTER DATABASE gis_eu RENAME TO gisold_eu;'
    psql -d postgres -c 'ALTER DATABASE gis_new RENAME TO gis_eu;'
    psql -d postgres -c 'ALTER DATABASE lowzoom_eu RENAME TO lowzoomold_eu;'
    psql -d postgres -c 'ALTER DATABASE lowzoom_new RENAME TO lowzoom_eu;'
 
    # Change date
    cd ~/fcmaps
    nano printmaps_webservice_capabilities.json

    # Restart server
    sync;sync;shutdown -r now
 
    # Check successfully restart
    who -b
 
    # Check database size
    psql -d postgres -c "SELECT pg_database.datname, pg_size_pretty(pg_database_size(pg_database.datname)) AS size FROM pg_database ORDER BY pg_database.datname;"
 
Contour data (as gis)
---------------------

    # Download contour data (Saarland)
    cd ~/data
    mkdir contours && cd contours
    pyhgtmap --area=6.24522242525:48.9348891599:7.83147757475:49.9660696535 --download-only --step=10 --start-node-id=1000000000 --start-way-id=1000000000 --max-nodes-per-tile=0 --no-zero-contour --pbf --srtm-user=... --srtm-password=...
        # https://pypi.org/project/pyhgtmap/
    
    # Create database *contours_new*
    createdb contours_new
    psql -d contours_new -c 'CREATE EXTENSION postgis;'
   
    # Load contour data into contour database (4h)
    osm2pgsql -d contours_new --slim --drop --flat-nodes=contournodes.dat --number-processes 10 --output pgsql --style ~/fcmaps/osmtools/contours.style lon*.pbf
        # https://osm2pgsql.org/doc/man/osm2pgsql-2-2-0.html

    # Activate new database *contours_eu*
    dropdb contoursold_eu
    psql -d postgres -c 'ALTER DATABASE contours_eu RENAME TO contoursold_eu;'
    psql -d postgres -c 'ALTER DATABASE contours_new RENAME TO contours_eu;'

    # Check database size
    psql -d postgres -c "SELECT pg_database.datname, pg_size_pretty(pg_database_size(pg_database.datname)) AS size FROM pg_database ORDER BY pg_database.datname;"

Test service (as gis)
---------------------

    # Start services manually
    cd ~/fcmaps
    nohup ./printmaps_buildservice 1>./logs/printmaps_buildservice.out 2>&1 &
    nohup ./printmaps_webservice 1>./logs/printmaps_webservice.out 2>&1 &

    # Test web service
    http://<server-dns-name>:8181/api/beta2/maps/capabilities/service
 
    # Search running service processes
    ps -Af | grep "printmaps_"
    
    # Terminate running service processes
    kill <pid>
 
    # Show log files
    cat ./logs/printmaps_webservice.log
    cat ./logs/printmaps_buildservice.log

Install services autostart (as root)
------------------------------------

    # Create service descriptions
    cd /etc/systemd/system
    mv /home/gis/fcmaps/fcmaps_build.service .
    mv /home/gis/fcmaps/fcmaps_web.service .

    # Enable autostart
    systemctl enable fcmaps_build.service
    systemctl enable fcmaps_web.service
 
    # Restart server
    sync;sync;shutdown -r now
 
    # Check successfully restart
    who -b

Backup & restore database
-------------------------
    
    # Backup
    cd ~/data/databases
    pg_dump contours_eu | gzip > contours_eu.gz
    
    # Restore
    createdb contours_new
    psql -d contours_new -c 'CREATE EXTENSION postgis;'
    gunzip -c contours_eu.gz | psql contours_new
    psql -d postgres -c 'ALTER DATABASE contours_new RENAME TO contours_eu;'

    # Check database size
    psql -d postgres -c "SELECT pg_database.datname, pg_size_pretty(pg_database_size(pg_database.datname)) AS size FROM pg_database ORDER BY pg_database.datname;"
